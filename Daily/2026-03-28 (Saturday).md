### MatLab + COMSOL proof of concept script:
```python
import com.comsol.model.*
import com.comsol.model.util.*

ModelUtil.clear;
model = ModelUtil.create('Model');
model.modelPath(pwd);
model.label('smiley_thermal_demo_fixed.mph');

%% PARAMETERS
model.param.set('Lx', '5[cm]');
model.param.set('Ly', '5[cm]');
model.param.set('Lz', '0.5[cm]');

model.param.set('T0', '293.15[K]');
model.param.set('t_pulse', '2[s]');
model.param.set('Etotal', '10[J]');

% Smiley geometry parameters
model.param.set('r_face', '1.4[cm]');
model.param.set('r_eye',  '0.22[cm]');
model.param.set('x_eye',  '0.55[cm]');
model.param.set('y_eye',  '0.45[cm]');
model.param.set('w_mouth','1.4[cm]');
model.param.set('y_mouth','-0.2[cm]');

% Material (HDPE approx)
model.param.set('rho', '950[kg/m^3]');
model.param.set('Cp',  '2300[J/(kg*K)]');
model.param.set('k',   '0.45[W/(m*K)]');

% Cooling
model.param.set('hconv', '8[W/(m^2*K)]');
model.param.set('Tamb',  '293.15[K]');

%% GEOMETRY
model.component.create('comp1', true);
model.component('comp1').geom.create('geom1', 3);
model.component('comp1').geom('geom1').lengthUnit('cm');

% Block
model.component('comp1').geom('geom1').create('blk1', 'Block');
model.component('comp1').geom('geom1').feature('blk1').set('size', {'Lx','Ly','Lz'});
model.component('comp1').geom('geom1').feature('blk1').set('base', 'center');

model.component('comp1').geom('geom1').run;

%% MATERIAL
model.component('comp1').material.create('mat1', 'Common');
model.component('comp1').material('mat1').selection.all;

model.component('comp1').material('mat1').propertyGroup('def').set('density', '950[kg/m^3]');
model.component('comp1').material('mat1').propertyGroup('def').set('heatcapacity', '2300[J/(kg*K)]');
model.component('comp1').material('mat1').propertyGroup('def').set( ...
    'thermalconductivity', {'0.45[W/(m*K)]','0','0','0','0.45[W/(m*K)]','0','0','0','0.45[W/(m*K)]'});

%% PHYSICS
model.component('comp1').physics.create('ht', 'HeatTransfer', 'geom1');
model.component('comp1').physics('ht').feature('init1').set('Tinit', 'T0');

% Convection everywhere
model.component('comp1').physics('ht').create('hf1','HeatFluxBoundary',2);
model.component('comp1').physics('ht').feature('hf1').selection.all;
model.component('comp1').physics('ht').feature('hf1').set('HeatFluxType','ConvectiveHeatFlux');
model.component('comp1').physics('ht').feature('hf1').set('h','hconv');
model.component('comp1').physics('ht').feature('hf1').set('Text','Tamb');

%% ANALYTIC FUNCTIONS

% Time pulse
model.component('comp1').func.create('pulse','Analytic');
model.component('comp1').func('pulse').set('funcname','pulsefun');
model.component('comp1').func('pulse').set('args','t');
model.component('comp1').func('pulse').set('expr','if(t<=t_pulse,1,0)');

%% SMILEY MASK (NO GEOMETRY NEEDED)

model.component('comp1').variable.create('var1');

% Face
model.component('comp1').variable('var1').set('mask_face', ...
    '(x^2 + y^2) < (r_face)^2');

% Eyes
model.component('comp1').variable('var1').set('mask_eyes', ...
    '((x+x_eye)^2+(y-y_eye)^2 < (r_eye)^2) || ((x-x_eye)^2+(y-y_eye)^2 < (r_eye)^2)');

% Mouth (rectangle band)
model.component('comp1').variable('var1').set('mask_mouth', ...
    '(abs(y - y_mouth) < 0.1[cm]) && (abs(x) < w_mouth/2)');

% Combined smiley
model.component('comp1').variable('var1').set('mask_smiley', ...
    '(mask_face && !mask_eyes) || mask_mouth');

%% HEAT INPUT

% Approximate heated area
model.param.set('Aheat', ...
    'pi*r_face^2 - 2*pi*r_eye^2 + w_mouth*0.2[cm]');

model.param.set('q0', 'Etotal/(Aheat*t_pulse)');

% Apply heat ONLY on top surface using mask
model.component('comp1').physics('ht').create('hf2','HeatFluxBoundary',2);

% Top boundary selection (robust box selection)
model.component('comp1').selection.create('top','Box');
model.component('comp1').selection('top').geom('geom1',2);
model.component('comp1').selection('top').set('zmin','Lz/2-1e-4[cm]');
model.component('comp1').selection('top').set('zmax','Lz/2+1e-4[cm]');

model.component('comp1').physics('ht').feature('hf2').selection.named('top');

model.component('comp1').physics('ht').feature('hf2').set( ...
    'q0','q0*pulsefun(t)*(mask_smiley)');

%% MESH
model.component('comp1').mesh.create('mesh1');
model.component('comp1').mesh('mesh1').autoMeshSize(3);
model.component('comp1').mesh('mesh1').run;

%% STUDY
model.study.create('std1');
model.study('std1').create('time','Transient');
model.study('std1').feature('time').set('tlist','range(0,0.05,5)');

%% SOLVE
model.sol.create('sol1');
model.sol('sol1').study('std1');
model.sol('sol1').attach('std1');
model.sol('sol1').runAll;

%% MAX TEMPERATURE
model.component('comp1').cpl.create('maxop','Maximum','geom1');
model.component('comp1').cpl('maxop').selection.all;

%% RESULTS

% 3D plot
model.result.create('pg1','PlotGroup3D');
model.result('pg1').create('surf1','Surface');
model.result('pg1').feature('surf1').set('expr','T');

% Max temperature plot
model.result.create('pg2','PlotGroup1D');
model.result('pg2').create('glob1','Global');
model.result('pg2').feature('glob1').set('expr','maxop(T)');

model.result('pg1').run;
model.result('pg2').run;

%% SAVE
mphsave(model,'smiley_thermal_demo_fixed.mph');

disp('Simulation complete.');
```


- builds geometry **entirely from code**
- creates mesh
- defines physics
- applies **smiley-shaped heating without fragile geometry**
- runs transient simulation
- produces:
    - 3D temperature field
    - max temperature vs time
- saves `.mph` file to open and view/scrub through from a MatLab window

>*note:* had to replace all material property assignments with explicit expressions for the solve error in matlab

> *note:* the repeated pulse version took probably like 10+ min in matlab and then almoat 14 min in COMSOL

> note: the 500 micron length surface + 10 nanosecond pulse system took  ~45-90 minutes to complete
